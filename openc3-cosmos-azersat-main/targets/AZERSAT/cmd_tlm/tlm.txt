TELEMETRY <%= target_name %> COMMS BIG_ENDIAN "Communications subsystem status"
  # APID 8: Communications telemetry packet
  # Contains antenna states, downlink status, signal strength, and power consumption
  <%= render "_ccsds_tlm.txt", locals: {apid: 8} %>

  # Antenna 1 (Primary) Status
  APPEND_ITEM ANT1_STATE 8 UINT "Antenna 1 deployment state"
    STATE STOWED 0 YELLOW      # Antenna is stowed (launch configuration)
    STATE DEPLOYING 1 YELLOW   # Antenna is currently deploying
    STATE DEPLOYED 2 GREEN     # Antenna is fully deployed and operational
    STATE STOWING 3 YELLOW     # Antenna is currently stowing
  APPEND_ITEM ANT1_TYPE 8 UINT "Antenna 1 type"
    STATE LOW_GAIN 0           # Omnidirectional, lower data rate
    STATE HIGH_GAIN 1          # Directional, higher data rate
  APPEND_ITEM ANT1_AZIMUTH 32 FLOAT "Antenna 1 azimuth angle"
    UNITS DEGREES DEG
    FORMAT_STRING "%0.1f"
  APPEND_ITEM ANT1_ELEVATION 32 FLOAT "Antenna 1 elevation angle"
    UNITS DEGREES DEG
    FORMAT_STRING "%0.1f"

  # Antenna 2 (Backup) Status
  APPEND_ITEM ANT2_STATE 8 UINT "Antenna 2 deployment state"
    STATE STOWED 0 YELLOW
    STATE DEPLOYING 1 YELLOW
    STATE DEPLOYED 2 GREEN
    STATE STOWING 3 YELLOW
  APPEND_ITEM ANT2_TYPE 8 UINT "Antenna 2 type"
    STATE LOW_GAIN 0
    STATE HIGH_GAIN 1
  APPEND_ITEM ANT2_AZIMUTH 32 FLOAT "Antenna 2 azimuth angle"
    UNITS DEGREES DEG
    FORMAT_STRING "%0.1f"
  APPEND_ITEM ANT2_ELEVATION 32 FLOAT "Antenna 2 elevation angle"
    UNITS DEGREES DEG
    FORMAT_STRING "%0.1f"

  # Downlink Operations
  APPEND_ITEM DOWNLINK_STATE 8 UINT "Downlink state"
    STATE IDLE 0 YELLOW        # No active downlink
    STATE ACTIVE 1 GREEN       # Actively transmitting data
  APPEND_ITEM DATA_RATE 8 UINT "Current data rate setting"
    STATE LOW 0                # Low data rate (10 Mbps)
    STATE MEDIUM 1             # Medium data rate (25 Mbps)
    STATE HIGH 2               # High data rate (75 Mbps)
  APPEND_ITEM DATA_BUFFER 32 FLOAT "Onboard data buffer percentage"
    UNITS PERCENT %
    FORMAT_STRING "%0.1f"
    LIMITS DEFAULT 1 ENABLED 0 5 95 100  # Warn if buffer nearly empty/full
  APPEND_ITEM SIGNAL_STRENGTH 32 FLOAT "Signal strength to ground station"
    UNITS DECIBELS DB
    FORMAT_STRING "%0.1f"
    LIMITS DEFAULT 1 ENABLED -80.0 -70.0 -30.0 -20.0  # Warn on weak signal
  APPEND_ITEM COMM_PWR 32 FLOAT "Communications subsystem power consumption"
    UNITS WATTS W
    FORMAT_STRING "%0.1f"
  APPEND_ITEM DOWNLINK_TIME_REMAINING 16 UINT "Downlink time remaining"
    UNITS SECONDS S

  # Derived timestamp field (standard for all packets)
  ITEM PACKET_TIME 0 0 DERIVED "Ruby time based on TIMESEC and TIMEUS"
    READ_CONVERSION unix_time_conversion.rb TIMESEC TIMEUS
